<site-modal id="myModal">
  <div class="modal-header">
    <h2>모달 제목</h2>
  </div>
  <div class="modal-body">
    <p>모달 내용</p>
  </div>

    <button onclick="document.getElementById('myModal').close()">닫기</button>
  </div>
</site-modal>
<button onclick="document.getElementById('myModal').open()">모달 열기</button>
